try { // Проверим, не превышает ли ставка баланс пользователя const user = await User.findById(userId); if (!user) return res.json({ error: 'Пользователь не найден' }); if (bet > user.balance) return res.json({ error: 'Недостаточно золота' }); // Определим maxPlayers для дурака (только 2) const maxPlayers = gameType === 'cards' ? 2 : 3; const room = new Room({ name, gameType, bet, maxPlayers, players: [userId] }); await room.save(); res.json(room); } catch (e) { res.json({ error: 'Ошибка создания комнаты' }); } }); // ---------- SOCKET.IO ---------- io.on('connection', (socket) => { console.log('New socket connection:', socket.id); let userId = null; socket.on('setUser', (id) => { userId = id; socket.data.userId = id; }); // Присоединение к комнате socket.on('joinRoom', async ({ roomId, userId }) => { try { const room = await Room.findById(roomId).populate('players'); state.results[userId] = value; // Находим следующего игрока, который ещё не бросил const players = state.players; let next = null; for (let p of players) { if (state.results[p._id] === undefined) { next = p._id; break; } } if (next) { state.currentTurn = next; state.statusText = `Ход игрока ${players.find(p => p._id === next).nickname}`; io.to(roomId).emit('gameState', state); } else { // Все бросили – определяем победителя let maxVal = -1, winners = []; const totalBet = room.bet * players.length; // каждый поставил ставку const winAmount = Math.floor(totalBet / winners.length); // Обновляем балансы победителей for (let w of winners) { const user = await User.findById(w._id); if (user) { user.balance += winAmount; await user.save(); } } // Возвращаем ставки проигравшим? Нет, они проигрывают свои ставки (они уже списаны при входе). // Спишем ставки у всех игроков (кроме победителей) – они уже были списаны при создании игры? // В нашей модели мы не списываем ставку сразу, будем считать, что ставка списывается при входе в игру. // Чтобы не усложнять, сделаем так: при старте игры списываем ставку у каждого. // Для простоты сейчас мы не списывали, поэтому сделаем здесь. // Итак, победители получают winAmount, проигравшие теряют свою ставку (у них уже списано). // Но мы не списывали ранее, поэтому списываем сейчас с проигравших. for (let p of players) { const user = await User.findById(p._id); if (!user) continue; if (winners.some(w => w._id === p._id)) { // победитель уже получил выигрыш выше } else { user.balance -= room.bet; await user.save(); } } // Обновляем состояние state.status = 'finished'; state.winner = winners[0]; // если несколько, берём первого state.winAmount = winAmount; state.statusText = `Игра завершена! Победитель: ${winners.map(w => w.nickname).join(', ')}`; io.to(roomId).emit('gameState', state); // Отправляем обновление баланса каждому игроку for (let p of players) { const user = await User.findById(p._id); if (user) { io.to(socket.id).emit('gameResult', { balance: user.balance }); // но это только текущему сокету, надо всем } } // Рассылаем всем в комнате обновление баланса for (let p of players) { const user = await User.findById(p._id); if (user) { io.to(roomId).emit('gameResult', { balance: user.balance, userId: p._id }); } } // Через некоторое время удаляем комнату или переводим в ожидание setTimeout(async () => { await Room.deleteOne({ _id: roomId }); gameStates.delete(roomId); io.emit('roomUpdate', await getAllWaitingRooms()); io.to(roomId).emit('roomDeleted'); }, 5000); } }); // Обработка выбора в монетке socket.on('coinChoice', async ({ roomId, userId, choice }) => { const state = gameStates.get(roomId); if (!state || state.gameType !== 'coin') return; if (state.status !== 'waiting_choice') return; if (state.currentTurn !== userId) return; state.choices[userId] = choice; // Находим следующего не выбравшего const players = state.players; let next = null; for (let p of players) { if (state.choices[p._id] === undefined) { next = p._id; break; } } if (next) { state.currentTurn = next; state.statusText = `Выбор игрока ${players.find(p => p._id === next).nickname}`; io.to(roomId).emit('gameState', state); } else { // Все выбрали – бросаем монету const result = Math.random() >= 0.5 ? 'heads' : 'tails'; state.result = result; // Определяем победителей const winners = players.filter(p => state.choices[p._id] === result); const room = await Room.findById(roomId); const totalBet = room.bet * players.length; const winAmount = winners.length > 0 ? Math.floor(totalBet / winners.length) : 0; // Обновляем балансы for (let p of players) { const user = await User.findById(p._id); if (!user) continue; if (winners.some(w => w._id === p._id)) { user.balance += winAmount; } else { user.balance -= room.bet; } await user.save(); } state.status = 'finished'; state.winners = winners; state.winAmount = winAmount; state.statusText = `Игра завершена! Выпал ${result === 'heads' ? 'Орёл' : 'Решка'}. Победители: ${winners.map(w => w.nickname).join(', ')}`; io.to(roomId).emit('gameState', state); // Рассылаем обновление баланса for (let p of players) { const user = await User.findById(p._id); if (user) { io.to(roomId).emit('gameResult', { balance: user.balance, userId: p._id }); } } // Удаляем комнату через 5 секунд setTimeout(async () => { await Room.deleteOne({ _id: roomId }); gameStates.delete(roomId); io.emit('roomUpdate', await getAllWaitingRooms()); io.to(roomId).emit('roomDeleted'); }, 5000); } }); // Отключение socket.on('disconnect', async () => { // Найдём комнаты, где был этот игрок, и удалим его if (socket.data.roomId) { const roomId = socket.data.roomId; try { const room = await Room.findById(roomId); io.to(roomId).emit('roomDeleted'); io.emit('roomUpdate', await getAllWaitingRooms()); } } } } catch (e) {} } }); }); // ---------- ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ ---------- async function getAllWaitingRooms() { const rooms = await Room.find({ status: 'waiting' }).populate('players', 'nickname'); return rooms; } async function getRoomData(roomId) { return await Room.findById(roomId).populate('players', 'nickname'); } function initGame(room) { const players = room.players; const gameType = room.gameType; const state = { gameType, players: players.map(p => ({ _id: p._id, nickname: p.nickname })), status: 'waiting_roll', // для костей results: {}, currentTurn: players[0]._id, statusText: `Ход игрока ${players[0].nickname}`, // для монетки choices: {}, result: null, winners: [], winAmount: 0, winner: null }; if (gameType === 'coin') { state.status = 'waiting_choice'; state.statusText = `Выбор игрока ${players[0].nickname}`; state.choices = {}; } return state; } // ---------- ЗАПУСК СЕРВЕРА ---------- const PORT = process.env.PORT || 3000;
